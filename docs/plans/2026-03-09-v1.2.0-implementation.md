# v1.2.0 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add Cloud Sync (self-hosted Postgres), interactive TUI browser, and Git-based team sharing to cortexmem.

**Architecture:** Three independent features extending the existing single-binary architecture. Cloud Sync adds a `src/cloud/` module with Postgres backend + sync engine that captures local mutations and replicates bidirectionally. TUI adds `src/tui/` using ratatui for an interactive memory browser reusing `CortexMemServer::call_*()`. Git Sync adds `src/sync/` for append-only chunk-based sharing via git repos.

**Tech Stack:** Rust, sqlx 0.8 (Postgres), jsonwebtoken 9, argon2 0.5, ratatui 0.29, crossterm 0.28, uuid 1

**Design doc:** `docs/plans/2026-03-09-roadmap-design.md` (Feature 9, 10, 11)

```rust
use anyhow::Result;
use super::Database;

#[derive(Debug, Clone)]
pub struct SyncMutation {
    pub seq: i64,
    pub entity: String,
    pub entity_key: String,
    pub op: String,
    pub payload: String,
    pub project: String,
    pub occurred_at: String,
    pub acked_at: Option<String>,
}

#[derive(Debug, Clone)]
pub struct SyncState {
    pub target_key: String,
    pub last_pushed_seq: i64,
    pub last_pulled_seq: i64,
    pub last_error: Option<String>,
    pub updated_at: String,
}

impl Database {
    pub fn insert_sync_mutation(
        &self,
        entity: &str,
        entity_key: &str,
        op: &str,
        payload: &str,
        project: &str,
    ) -> Result<i64> {
        self.conn().execute(
            "INSERT INTO sync_mutations (entity, entity_key, op, payload, project)
             VALUES (?1, ?2, ?3, ?4, ?5)",
            rusqlite::params![entity, entity_key, op, payload, project],
        )?;
        Ok(self.conn().last_insert_rowid())
    }

    pub fn list_unacked_mutations(&self, limit: i64) -> Result<Vec<SyncMutation>> {
        let mut stmt = self.conn().prepare(
            "SELECT seq, entity, entity_key, op, payload, project, occurred_at, acked_at
             FROM sync_mutations WHERE acked_at IS NULL ORDER BY seq ASC LIMIT ?1",
        )?;
        let rows = stmt.query_map([limit], |row| {
            Ok(SyncMutation {
                seq: row.get(0)?,
                entity: row.get(1)?,
                entity_key: row.get(2)?,
                op: row.get(3)?,
                payload: row.get(4)?,
                project: row.get(5)?,
                occurred_at: row.get(6)?,
                acked_at: row.get(7)?,
            })
        })?;
        Ok(rows.collect::<Result<Vec<_>, _>>()?)
    }

    pub fn ack_mutations(&self, up_to_seq: i64) -> Result<()> {
        self.conn().execute(
            "UPDATE sync_mutations SET acked_at = datetime('now') WHERE seq <= ?1 AND acked_at IS NULL",
            [up_to_seq],
        )?;
        Ok(())
    }

    pub fn update_sync_state(
        &self,
        target_key: &str,
        last_pushed_seq: i64,
        last_pulled_seq: i64,
        last_error: Option<&str>,
    ) -> Result<()> {
        self.conn().execute(
            "INSERT INTO sync_state (target_key, last_pushed_seq, last_pulled_seq, last_error, updated_at)
             VALUES (?1, ?2, ?3, ?4, datetime('now'))
             ON CONFLICT(target_key) DO UPDATE SET
                last_pushed_seq = ?2, last_pulled_seq = ?3, last_error = ?4, updated_at = datetime('now')",
            rusqlite::params![target_key, last_pushed_seq, last_pulled_seq, last_error],
        )?;
        Ok(())
    }

    pub fn get_sync_state(&self, target_key: &str) -> Result<Option<SyncState>> {
        let mut stmt = self.conn().prepare(
            "SELECT target_key, last_pushed_seq, last_pulled_seq, last_error, updated_at
             FROM sync_state WHERE target_key = ?1",
        )?;
        let result = stmt.query_row([target_key], |row| {
            Ok(SyncState {
                target_key: row.get(0)?,
                last_pushed_seq: row.get(1)?,
                last_pulled_seq: row.get(2)?,
                last_error: row.get(3)?,
                updated_at: row.get(4)?,
            })
        });
        match result {
            Ok(state) => Ok(Some(state)),
            Err(rusqlite::Error::QueryReturnedNoRows) => Ok(None),
            Err(e) => Err(e.into()),
        }
    }

    pub fn record_sync_chunk(&self, chunk_id: &str) -> Result<bool> {
        let result = self.conn().execute(
            "INSERT OR IGNORE INTO sync_chunks (chunk_id) VALUES (?1)",
            [chunk_id],
        )?;
        Ok(result > 0)
    }
}
```

```rust
use std::sync::Arc;
use anyhow::Result;
use axum::{Router, Json, extract::State};
use sqlx::PgPool;
use serde_json::json;

pub struct CloudState {
    pub pool: PgPool,
    pub jwt_secret: String,
}

pub type SharedCloudState = Arc<CloudState>;

pub fn build_cloud_router(state: SharedCloudState) -> Router {
    Router::new()
        .route("/health", axum::routing::get(health))
        .with_state(state)
}

async fn health(State(state): State<SharedCloudState>) -> Json<serde_json::Value> {
    let db_ok = sqlx::query("SELECT 1")
        .fetch_one(&state.pool)
        .await
        .is_ok();
    Json(json!({
        "status": if db_ok { "ok" } else { "degraded" },
        "version": env!("CARGO_PKG_VERSION"),
    }))
}

pub async fn start_cloud_server(
    database_url: &str,
    jwt_secret: &str,
    host: &str,
    port: u16,
) -> Result<()> {
    let pool = PgPool::connect(database_url).await?;
    super::schema::run_migrations(&pool).await?;

    let state = Arc::new(CloudState {
        pool,
        jwt_secret: jwt_secret.to_string(),
    });

    let app = build_cloud_router(state);
    let listener = tokio::net::TcpListener::bind(format!("{host}:{port}")).await?;
    tracing::info!("Cloud server listening on {host}:{port}");
    axum::serve(listener, app).await?;
    Ok(())
}
```

```rust
use anyhow::{Result, bail};
use argon2::{Argon2, PasswordHasher, PasswordVerifier, password_hash::{SaltString, rand_core::OsRng}};
use jsonwebtoken::{encode, decode, Header, Validation, EncodingKey, DecodingKey};
use serde::{Serialize, Deserialize};
use sqlx::PgPool;
use uuid::Uuid;

#[derive(Debug, Serialize, Deserialize)]
pub struct Claims {
    pub sub: String,        // account_id
    pub exp: usize,
    pub iat: usize,
}

#[derive(Debug, Deserialize)]
pub struct RegisterRequest {
    pub email: String,
    pub password: String,
}

#[derive(Debug, Deserialize)]
pub struct LoginRequest {
    pub email: String,
    pub password: String,
}

#[derive(Debug, Serialize)]
pub struct AuthResponse {
    pub token: String,
    pub account_id: String,
}

#[derive(Debug, Serialize)]
pub struct ApiKeyResponse {
    pub key: String,     // Full key (only shown once): ctx_...
    pub prefix: String,  // First 8 chars for identification
}

pub fn hash_password(password: &str) -> Result<String> {
    let salt = SaltString::generate(&mut OsRng);
    let hash = Argon2::default()
        .hash_password(password.as_bytes(), &salt)
        .map_err(|e| anyhow::anyhow!("Failed to hash password: {e}"))?;
    Ok(hash.to_string())
}

pub fn verify_password(password: &str, hash: &str) -> Result<bool> {
    let parsed = argon2::PasswordHash::new(hash)
        .map_err(|e| anyhow::anyhow!("Invalid password hash: {e}"))?;
    Ok(Argon2::default().verify_password(password.as_bytes(), &parsed).is_ok())
}

pub fn create_jwt(account_id: &Uuid, secret: &str) -> Result<String> {
    let now = chrono::Utc::now().timestamp() as usize;
    let claims = Claims {
        sub: account_id.to_string(),
        iat: now,
        exp: now + 86400 * 7, // 7 days
    };
    let token = encode(
        &Header::default(),
        &claims,
        &EncodingKey::from_secret(secret.as_bytes()),
    )?;
    Ok(token)
}

pub fn verify_jwt(token: &str, secret: &str) -> Result<Claims> {
    let data = decode::<Claims>(
        token,
        &DecodingKey::from_secret(secret.as_bytes()),
        &Validation::default(),
    )?;
    Ok(data.claims)
}

pub async fn register(pool: &PgPool, email: &str, password: &str) -> Result<Uuid> {
    if password.len() < 8 {
        bail!("Password must be at least 8 characters");
    }
    let password_hash = hash_password(password)?;
    let row: (Uuid,) = sqlx::query_as(
        "INSERT INTO accounts (email, password_hash) VALUES ($1, $2) RETURNING id",
    )
    .bind(email)
    .bind(&password_hash)
    .fetch_one(pool)
    .await?;
    Ok(row.0)
}

pub async fn login(pool: &PgPool, email: &str, password: &str, jwt_secret: &str) -> Result<AuthResponse> {
    let row: Option<(Uuid, String)> = sqlx::query_as(
        "SELECT id, password_hash FROM accounts WHERE email = $1",
    )
    .bind(email)
    .fetch_optional(pool)
    .await?;

    let (account_id, hash) = row.ok_or_else(|| anyhow::anyhow!("Invalid credentials"))?;

    if !verify_password(password, &hash)? {
        bail!("Invalid credentials");
    }

    let token = create_jwt(&account_id, jwt_secret)?;
    Ok(AuthResponse {
        token,
        account_id: account_id.to_string(),
    })
}

pub async fn generate_api_key(pool: &PgPool, account_id: &Uuid) -> Result<ApiKeyResponse> {
    let raw_key = format!("ctx_{}", Uuid::new_v4().to_string().replace('-', ""));
    let prefix = &raw_key[..12];
    let key_hash = hash_password(&raw_key)?;

    sqlx::query(
        "INSERT INTO api_keys (account_id, key_hash, prefix) VALUES ($1, $2, $3)",
    )
    .bind(account_id)
    .bind(&key_hash)
    .bind(prefix)
    .fetch_optional(pool)
    .await?;

    Ok(ApiKeyResponse {
        key: raw_key,
        prefix: prefix.to_string(),
    })
}

pub async fn verify_api_key(pool: &PgPool, key: &str) -> Result<Uuid> {
    if key.len() < 12 {
        bail!("Invalid API key format");
    }
    let prefix = &key[..12];

    let rows: Vec<(Uuid, String)> = sqlx::query_as(
        "SELECT account_id, key_hash FROM api_keys WHERE prefix = $1 AND revoked_at IS NULL",
    )
    .bind(prefix)
    .fetch_all(pool)
    .await?;

    for (account_id, hash) in rows {
        if verify_password(key, &hash)? {
            return Ok(account_id);
        }
    }
    bail!("Invalid API key")
}
```

```rust
pub async fn push_mutations(
    pool: &PgPool,
    account_id: &Uuid,
    req: &PushRequest,
) -> Result<PushResponse> {
    let mut last_seq: i64 = 0;
    let mut accepted: i64 = 0;

    for m in &req.mutations {
        // Verify project is enrolled
        let enrolled: bool = sqlx::query_scalar(
            "SELECT EXISTS(SELECT 1 FROM enrolled_projects WHERE account_id = $1 AND project = $2)",
        )
        .bind(account_id)
        .bind(&m.project)
        .fetch_one(pool)
        .await?;

        if !enrolled {
            continue; // Skip unenrolled projects
        }

        let row: (i64,) = sqlx::query_as(
            "INSERT INTO sync_mutations (account_id, entity, entity_key, op, payload, project, occurred_at)
             VALUES ($1, $2, $3, $4, $5, $6, $7::timestamptz)
             RETURNING seq",
        )
        .bind(account_id)
        .bind(&m.entity)
        .bind(&m.entity_key)
        .bind(&m.op)
        .bind(&m.payload)
        .bind(&m.project)
        .bind(&m.occurred_at)
        .fetch_one(pool)
        .await?;

        last_seq = row.0;
        accepted += 1;
    }

    Ok(PushResponse { accepted, last_seq })
}

pub async fn pull_mutations(
    pool: &PgPool,

        .bind(since_seq)
        .bind(proj)
        .bind(limit)
        .fetch_all(pool)
        .await?
    } else {
        sqlx::query_as::<_, (i64, String, String, String, serde_json::Value, String, String)>(

        .bind(since_seq)
        .bind(limit)

        .await?
```

```rust
use anyhow::Result;
use serde::{Serialize, Deserialize};
use crate::db::Database;

#[derive(Debug)]
pub struct SyncConfig {
    pub server_url: String,
    pub api_key: String,
}

#[derive(Debug, Serialize)]
struct PushPayload {
    mutations: Vec<MutationPayload>,
}

#[derive(Debug, Serialize)]
struct MutationPayload {
    entity: String,
    entity_key: String,
    op: String,
    payload: serde_json::Value,
    project: String,
    occurred_at: String,
}

#[derive(Debug, Deserialize)]
struct PushResponse {
    accepted: i64,
    last_seq: i64,
}

#[derive(Debug, Deserialize)]
struct PullMutation {
    seq: i64,
    entity: String,
    entity_key: String,
    op: String,
    payload: serde_json::Value,
    project: String,
    occurred_at: String,
}

pub async fn push(db: &Database, config: &SyncConfig) -> Result<i64> {
    let mutations = db.list_unacked_mutations(100)?;
    if mutations.is_empty() {
        return Ok(0);
    }

    let payload = PushPayload {
        mutations: mutations.iter().map(|m| MutationPayload {
            entity: m.entity.clone(),
            entity_key: m.entity_key.clone(),
            op: m.op.clone(),
            payload: serde_json::from_str(&m.payload).unwrap_or(serde_json::Value::Null),
            project: m.project.clone(),
            occurred_at: m.occurred_at.clone(),
        }).collect(),
    };

    let client = reqwest::Client::new();
    let resp = client
        .post(format!("{}/sync/push", config.server_url))
        .bearer_auth(&config.api_key)
        .json(&payload)
        .send()
        .await?;

    if !resp.status().is_success() {
        let status = resp.status();
        let body = resp.text().await.unwrap_or_default();
        anyhow::bail!("Push failed ({status}): {body}");
    }

    let result: PushResponse = resp.json().await?;

    // Ack locally
    if let Some(last) = mutations.last() {
        db.ack_mutations(last.seq)?;
    }

    // Update sync state
    let state = db.get_sync_state("cloud:default")?;
    let pulled = state.as_ref().map_or(0, |s| s.last_pulled_seq);
    db.update_sync_state("cloud:default", result.last_seq, pulled, None)?;

    Ok(result.accepted)
}

pub async fn pull(db: &Database, config: &SyncConfig) -> Result<i64> {
    let state = db.get_sync_state("cloud:default")?;
    let since_seq = state.as_ref().map_or(0, |s| s.last_pulled_seq);

    let client = reqwest::Client::new();
    let resp = client
        .get(format!("{}/sync/pull", config.server_url))
        .bearer_auth(&config.api_key)
        .query(&[("since_seq", since_seq.to_string())])
        .send()
        .await?;

    if !resp.status().is_success() {
        let status = resp.status();
        let body = resp.text().await.unwrap_or_default();
        anyhow::bail!("Pull failed ({status}): {body}");
    }

    let mutations: Vec<PullMutation> = resp.json().await?;
    let count = mutations.len() as i64;

    let mut max_seq = since_seq;
    for m in &mutations {
        apply_remote_mutation(db, m)?;
        if m.seq > max_seq {
            max_seq = m.seq;
        }
    }

    // Update sync state
    let pushed = state.as_ref().map_or(0, |s| s.last_pushed_seq);
    db.update_sync_state("cloud:default", pushed, max_seq, None)?;

    Ok(count)
}

fn apply_remote_mutation(db: &Database, m: &PullMutation) -> Result<()> {
    match (m.entity.as_str(), m.op.as_str()) {
        ("observation", "insert" | "upsert") => {
            // Apply as import — uses content hash dedup
            tracing::info!("Applying remote {} for observation {}", m.op, m.entity_key);
            // The actual application logic will use db.import_observation()
            // which already handles dedup by content_hash
            if let Ok(obs) = serde_json::from_value::<crate::db::observations::Observation>(m.payload.clone()) {
                let _ = db.import_observation(&obs);
            }
        }
        ("observation", "soft_delete") => {
            if let Ok(id) = m.entity_key.parse::<i64>() {
                let _ = db.soft_delete(id);
            }
        }
        ("observation", "hard_delete") => {
            if let Ok(id) = m.entity_key.parse::<i64>() {
                let _ = db.hard_delete(id);
            }
        }
        _ => {
            tracing::warn!("Unknown mutation: {} {}", m.entity, m.op);
        }
    }
    Ok(())
}

pub async fn sync_once(db: &Database, config: &SyncConfig) -> Result<(i64, i64)> {
    let pushed = push(db, config).await?;
    let pulled = pull(db, config).await?;
    Ok((pushed, pulled))
}
```

```rust
pub fn run(server: Arc<CortexMemServer>) -> Result<()> {
    terminal::enable_raw_mode()?;
    let mut stdout = io::stdout();
    crossterm::execute!(stdout, terminal::EnterAlternateScreen)?;
    let backend = CrosstermBackend::new(stdout);
    let mut terminal = Terminal::new(backend)?;

    let mut app = App::new(server);

        terminal.draw(|f| render(&app, f))?;

        if event::poll(std::time::Duration::from_millis(100))? {

                handle_input(&mut app, key);

        }

        if app.should_quit {

        }
    }

    terminal::disable_raw_mode()?;
    crossterm::execute!(terminal.backend_mut(), terminal::LeaveAlternateScreen)?;
    Ok(())
}
```

```rust
pub fn handle_input(app: &mut App, key: KeyEvent) {
    let Screen::Search { query, cursor } = &mut app.screen else { return };

    match key.code {
        KeyCode::Char(c) => {
            query.insert(*cursor, c);
            *cursor += 1;
        }
        KeyCode::Backspace => {
            if *cursor > 0 {
                *cursor -= 1;
                query.remove(*cursor);
            }
        }
        KeyCode::Esc => {
            app.pop_screen();
        }
        KeyCode::Enter => {
            if !query.is_empty() {
                let q = query.clone();
                let results = app.server.call_search(&q, None, None, None, Some(50));
                app.screen = Screen::SearchResults {
                    query: q,
                    results,
                    selected: 0,
                };
            }
        }
        _ => {}
    }
}

pub fn render_results(f: &mut Frame, app: &App) {
    let Screen::SearchResults { query, results, selected } = &app.screen else { return };

    let chunks = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Length(3),
            Constraint::Min(1),

        .split(f.area());

        .style(Style::default().fg(theme::BLUE).add_modifier(Modifier::BOLD))
        .block(Block::default().borders(Borders::ALL).border_style(Style::default().fg(theme::SURFACE1)));
    f.render_widget(header, chunks[0]);

    let items: Vec<ratatui::widgets::ListItem> = results.iter().enumerate().map(|(i, r)| {
        let style = if i == *selected {
            Style::default().fg(theme::BASE).bg(theme::BLUE)
        } else {
            Style::default().fg(theme::TEXT)
        };
        let concepts = r.concepts.as_ref()
            .map(|c| c.join(", "))
            .unwrap_or_default();
        ratatui::widgets::ListItem::new(format!(
            "  [{}] {} ({}) — {:.2}  {}",
            r.id, r.title, r.obs_type, r.score, concepts
        )).style(style)
    }).collect();

    let list = ratatui::widgets::List::new(items)
        .block(Block::default()
            .title(" ↑/↓ navigate, Enter detail, Esc back ")

pub fn handle_results_input(app: &mut App, key: KeyEvent) {
    let Screen::SearchResults { results, selected, .. } = &mut app.screen else { return };

    match key.code {
        KeyCode::Up | KeyCode::Char('k') => {
            if *selected > 0 { *selected -= 1; }
        }
        KeyCode::Down | KeyCode::Char('j') => {
            if *selected + 1 < results.len() { *selected += 1; }
        }
        KeyCode::Enter => {
            if let Some(r) = results.get(*selected) {
                if let Ok(Some(obs)) = app.server.call_get(r.id) {
                    app.push_screen(Screen::ObservationDetail { obs, scroll: 0 });
                }
            }
        }
        KeyCode::Esc => {
            app.pop_screen();
        }
        KeyCode::Char('q') => {
            app.should_quit = true;
        }
        _ => {}
    }
}
```

```rust
use anyhow::Result;
use serde::{Serialize, Deserialize};
use crate::db::Database;
use crate::db::observations::Observation;
use crate::db::sessions::Session;

#[derive(Debug, Serialize, Deserialize)]
pub struct SyncChunk {
    pub chunk_id: String,
    pub source: String,
    pub project: String,
    pub exported_at: String,
    pub observations: Vec<Observation>,
    pub sessions: Vec<Session>,
}

pub fn create_chunk(db: &Database, project: Option<&str>) -> Result<SyncChunk> {
    let observations = db.list_all_observations_for_export(project)?;
    let sessions = db.list_all_sessions_for_export(project)?;
    let hostname = hostname::get()
        .map(|h| h.to_string_lossy().to_string())
        .unwrap_or_else(|_| "unknown".to_string());

    Ok(SyncChunk {
        chunk_id: uuid::Uuid::new_v4().to_string(),
        source: hostname,
        project: project.unwrap_or("all").to_string(),
        exported_at: chrono::Utc::now().to_rfc3339(),
        observations,
        sessions,
    })
}

pub fn import_chunk(db: &Database, json: &str) -> Result<usize> {
    let chunk: SyncChunk = serde_json::from_str(json)?;

    // Check if chunk was already imported
    if !db.record_sync_chunk(&chunk.chunk_id)? {
        return Ok(0); // Already imported
    }

    let mut count = 0;
    for obs in &chunk.observations {
        if db.import_observation(obs)? {
            count += 1;
        }
    }
    Ok(count)
}

/// Performs a full git sync cycle:
/// 1. Create chunk from local data
/// 2. Write to sync repo as JSON file
/// 3. Git commit + push
/// 4. Git pull
/// 5. Import any new chunks
pub fn sync_via_git(db: &Database, sync_dir: &std::path::Path, project: &str) -> Result<(usize, usize)> {
    use std::process::Command;
    use std::fs;

    let chunks_dir = sync_dir.join("chunks").join(project);
    fs::create_dir_all(&chunks_dir)?;

    // 1. Export
    let chunk = create_chunk(db, Some(project))?;
    let chunk_file = chunks_dir.join(format!("{}.json", chunk.chunk_id));
    fs::write(&chunk_file, serde_json::to_string_pretty(&chunk)?)?;
    let exported = chunk.observations.len();

    // 2. Git add + commit + push
    Command::new("git")
        .args(["add", "."])
        .current_dir(sync_dir)
        .output()?;

    let commit_result = Command::new("git")
        .args(["commit", "-m", &format!("sync: {project} chunk {}", chunk.chunk_id)])
        .current_dir(sync_dir)
        .output()?;

    if commit_result.status.success() {
        Command::new("git")
            .args(["push"])
            .current_dir(sync_dir)
            .output()?;
    }

    // 3. Git pull
    Command::new("git")
        .args(["pull", "--rebase"])
        .current_dir(sync_dir)
        .output()?;

    // 4. Import new chunks
    let mut imported = 0;
    for entry in fs::read_dir(&chunks_dir)? {
        let entry = entry?;
        if entry.path().extension().is_some_and(|e| e == "json") {
            let content = fs::read_to_string(entry.path())?;
            imported += import_chunk(db, &content)?;
        }
    }

    Ok((exported, imported))
}

pub fn init_sync_repo(sync_dir: &std::path::Path, remote_url: Option<&str>) -> Result<()> {
    use std::process::Command;
    use std::fs;

    if sync_dir.join(".git").exists() {
        return Ok(()); // Already initialized
    }

    fs::create_dir_all(sync_dir)?;

    if let Some(url) = remote_url {
        Command::new("git")
            .args(["clone", url, "."])
            .current_dir(sync_dir)
            .output()?;
    } else {
        Command::new("git")
            .args(["init"])
            .current_dir(sync_dir)
            .output()?;
    }

    Ok(())
}
```

```rust
use crate::sync::git;

fn default_sync_dir() -> std::path::PathBuf {

        .join("cortexmem")
        .join("sync")

pub fn run_sync_init(repo: Option<&str>, path: Option<&std::path::Path>) -> Result<()> {
    let sync_dir = path.map(|p| p.to_path_buf()).unwrap_or_else(default_sync_dir);
    git::init_sync_repo(&sync_dir, repo)?;
    println!("Sync repo initialized at {}", sync_dir.display());
    Ok(())
}

pub fn run_sync(db: &crate::db::Database, project: Option<&str>) -> Result<()> {
    let sync_dir = default_sync_dir();
    let project = project.unwrap_or(&crate::cli::detect_project());
    let (exported, imported) = git::sync_via_git(db, &sync_dir, project)?;
    println!("Sync complete: exported {exported} observations, imported {imported} new");
    Ok(())
}

pub fn run_sync_status(db: &crate::db::Database) -> Result<()> {
    let state = db.get_sync_state("git:default")?;
    match state {
        Some(s) => {
            println!("Last sync: {}", s.updated_at);
            println!("Last pushed seq: {}", s.last_pushed_seq);
            println!("Last pulled seq: {}", s.last_pulled_seq);
            if let Some(err) = s.last_error {
                println!("Last error: {err}");
            }
        }
        None => println!("No sync history found. Run `cortexmem sync init` first."),
    }
    Ok(())
}

pub async fn run_sync_auto(db: &crate::db::Database, interval: u64, project: Option<&str>) -> Result<()> {
    let sync_dir = default_sync_dir();
    let project = project.unwrap_or("all").to_string();
    println!("Starting auto-sync every {interval}s for project: {project}");

    loop {
        match git::sync_via_git(db, &sync_dir, &project) {
            Ok((exp, imp)) => {
                if exp > 0 || imp > 0 {
                    println!("[{}] Synced: +{exp} exported, +{imp} imported",
                        chrono::Local::now().format("%H:%M:%S"));
                }
            }
            Err(e) => {
                eprintln!("[{}] Sync error: {e}",
                    chrono::Local::now().format("%H:%M:%S"));
            }
        }
        tokio::time::sleep(std::time::Duration::from_secs(interval)).await;
    }
}
```
