**Step 2: Create the auto-update workflow in homebrew-tap**

Create `.github/workflows/update-formula.yml`:

```yaml
name: Update Formula

on:
  repository_dispatch:
    types: [cortexmem-release]
  workflow_dispatch:
    inputs:
      version:
        description: 'Release version (e.g. 1.3.0)'
        required: true

permissions:
  contents: write

jobs:
  update:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set version
        id: version
        run: |
          VERSION="${{ github.event.client_payload.version || github.event.inputs.version }}"
          echo "version=$VERSION" >> "$GITHUB_OUTPUT"

      - name: Download and compute SHA256
        env:
          VERSION: ${{ steps.version.outputs.version }}
        run: |
          BASE="https://github.com/pablocalofatti/cortexmem/releases/download/v${VERSION}"
          for ARCHIVE in cortexmem-darwin-arm64.tar.gz cortexmem-darwin-x64.tar.gz cortexmem-linux-arm64.tar.gz cortexmem-linux-x64.tar.gz; do
            curl -fsSL "${BASE}/${ARCHIVE}" -o "${ARCHIVE}"
            SHA=$(sha256sum "${ARCHIVE}" | cut -d' ' -f1)
            VARNAME=$(echo "${ARCHIVE}" | sed 's/cortexmem-//;s/\.tar\.gz//;s/-/_/g' | tr '[:lower:]' '[:upper:]')
            echo "SHA_${VARNAME}=${SHA}" >> "$GITHUB_ENV"
          done

      - name: Update formula
        env:
          VERSION: ${{ steps.version.outputs.version }}
        run: |
          sed -i "s/version \".*\"/version \"${VERSION}\"/" Formula/cortexmem.rb
          sed -i "s/PLACEHOLDER_DARWIN_ARM64/${SHA_DARWIN_ARM64}/" Formula/cortexmem.rb
          sed -i "s/PLACEHOLDER_DARWIN_X64/${SHA_DARWIN_X64}/" Formula/cortexmem.rb
          sed -i "s/PLACEHOLDER_LINUX_ARM64/${SHA_LINUX_ARM64}/" Formula/cortexmem.rb
          sed -i "s/PLACEHOLDER_LINUX_X64/${SHA_LINUX_X64}/" Formula/cortexmem.rb

      - name: Commit and push
        env:
          VERSION: ${{ steps.version.outputs.version }}
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add Formula/cortexmem.rb
          git commit -m "Update cortexmem to v${VERSION}"
          git push
```

**Step 3: Create the install script**

Create `npm/scripts/install.js`:

```js
"use strict";

const https = require("https");
const fs = require("fs");
const path = require("path");
const { execFileSync } = require("child_process");

const VERSION = require("../package.json").version;
const BASE_URL = `https://github.com/pablocalofatti/cortexmem/releases/download/v${VERSION}`;

const PLATFORMS = {
  "darwin-arm64": "cortexmem-darwin-arm64.tar.gz",
  "darwin-x64": "cortexmem-darwin-x64.tar.gz",
  "linux-arm64": "cortexmem-linux-arm64.tar.gz",
  "linux-x64": "cortexmem-linux-x64.tar.gz",
};

function getPlatformKey() {
  const platform = process.platform;
  const arch = process.arch;

  const normalizedPlatform = platform === "darwin" ? "darwin" : "linux";
  const normalizedArch = arch === "arm64" ? "arm64" : "x64";

  return `${normalizedPlatform}-${normalizedArch}`;
}

function download(url) {
  return new Promise((resolve, reject) => {
    https.get(url, (res) => {
      if (res.statusCode >= 300 && res.statusCode < 400 && res.headers.location) {
        return download(res.headers.location).then(resolve).catch(reject);
      }
      if (res.statusCode !== 200) {
        return reject(new Error(`Download failed: HTTP ${res.statusCode}`));
      }
      const chunks = [];
      res.on("data", (chunk) => chunks.push(chunk));
      res.on("end", () => resolve(Buffer.concat(chunks)));
      res.on("error", reject);
    }).on("error", reject);
  });
}

async function main() {
  const key = getPlatformKey();
  const archive = PLATFORMS[key];

  if (!archive) {
    console.error(`Unsupported platform: ${process.platform}-${process.arch}`);
    console.error(`Supported: ${Object.keys(PLATFORMS).join(", ")}`);
    process.exit(1);
  }

  const url = `${BASE_URL}/${archive}`;
  console.log(`Downloading cortexmem v${VERSION} for ${key}...`);

  const tarball = await download(url);

  const tmpFile = path.join(__dirname, "..", "bin", `_${archive}`);
  const binDir = path.join(__dirname, "..", "bin");

  fs.mkdirSync(binDir, { recursive: true });
  fs.writeFileSync(tmpFile, tarball);

  execFileSync("tar", ["xzf", tmpFile, "-C", binDir], { stdio: "inherit" });
  fs.unlinkSync(tmpFile);

  const binaryPath = path.join(binDir, "cortexmem");
  fs.chmodSync(binaryPath, 0o755);

  console.log(`cortexmem v${VERSION} installed successfully.`);
}

main().catch((err) => {
  console.error("Failed to install cortexmem:", err.message);
  process.exit(1);
});
```

### Task 4: Shell installer script

**Files:**
- Create: `scripts/install.sh`

**Step 1: Create the installer**

Create `scripts/install.sh`:

```bash
#!/usr/bin/env bash
set -euo pipefail

REPO="pablocalofatti/cortexmem"
INSTALL_DIR="${CORTEXMEM_INSTALL_DIR:-$HOME/.cortexmem/bin}"

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

info()  { echo -e "${GREEN}info${NC}: $*"; }
warn()  { echo -e "${YELLOW}warn${NC}: $*"; }
error() { echo -e "${RED}error${NC}: $*" >&2; exit 1; }

# Detect platform
detect_platform() {
  local os arch
  os="$(uname -s)"
  arch="$(uname -m)"

  case "$os" in
    Darwin) os="darwin" ;;
    Linux)  os="linux" ;;
    *)      error "Unsupported OS: $os" ;;
  esac

  case "$arch" in
    x86_64|amd64)  arch="x64" ;;
    aarch64|arm64) arch="arm64" ;;
    *)             error "Unsupported architecture: $arch" ;;
  esac

  echo "${os}-${arch}"
}

# Fetch latest version from GitHub API
get_latest_version() {
  local url="https://api.github.com/repos/${REPO}/releases/latest"
  local tag

  if command -v curl >/dev/null 2>&1; then
    tag=$(curl -fsSL "$url" | grep '"tag_name"' | head -1 | cut -d'"' -f4)
  elif command -v wget >/dev/null 2>&1; then
    tag=$(wget -qO- "$url" | grep '"tag_name"' | head -1 | cut -d'"' -f4)
  else
    error "Neither curl nor wget found. Install one and try again."
  fi

  echo "${tag#v}"
}

# Download file
download() {
  local url="$1" dest="$2"
  if command -v curl >/dev/null 2>&1; then
    curl -fsSL "$url" -o "$dest"
  else
    wget -qO "$dest" "$url"
  fi
}

main() {
  local platform version archive url

  info "Detecting platform..."
  platform="$(detect_platform)"
  info "Platform: ${platform}"

  info "Fetching latest version..."
  version="$(get_latest_version)"
  info "Version: v${version}"

  archive="cortexmem-${platform}.tar.gz"
  url="https://github.com/${REPO}/releases/download/v${version}/${archive}"

  info "Downloading ${archive}..."
  local tmp
  tmp="$(mktemp -d)"
  download "$url" "${tmp}/${archive}"

  info "Installing to ${INSTALL_DIR}..."
  mkdir -p "$INSTALL_DIR"
  tar xzf "${tmp}/${archive}" -C "$INSTALL_DIR"
  chmod +x "${INSTALL_DIR}/cortexmem"
  rm -rf "$tmp"

  # Add to PATH if not already there
  if ! echo "$PATH" | tr ':' '\n' | grep -qx "$INSTALL_DIR"; then
    local shell_rc=""
    if [ -f "$HOME/.zshrc" ]; then
      shell_rc="$HOME/.zshrc"
    elif [ -f "$HOME/.bashrc" ]; then
      shell_rc="$HOME/.bashrc"
    fi

    if [ -n "$shell_rc" ]; then
      local path_line="export PATH=\"${INSTALL_DIR}:\$PATH\""
      if ! grep -qF "$INSTALL_DIR" "$shell_rc" 2>/dev/null; then
        echo "" >> "$shell_rc"
        echo "# cortexmem" >> "$shell_rc"
        echo "$path_line" >> "$shell_rc"
        warn "Added ${INSTALL_DIR} to PATH in ${shell_rc}. Restart your shell or run:"
        echo "  source ${shell_rc}"
      fi
    else
      warn "Add ${INSTALL_DIR} to your PATH manually."
    fi
  fi

  info "cortexmem v${version} installed successfully!"
  "${INSTALL_DIR}/cortexmem" --version
  echo ""
  info "Next: run 'cortexmem setup' to configure your AI agent."
  info "Uninstall: rm -rf ${INSTALL_DIR} && remove the PATH line from your shell config."
}

main
```

**Step 3: Create `src/cli/doctor.rs`**

```rust
use crate::mcp::CortexMemServer;

const CURRENT_SCHEMA_VERSION: i64 = 3;

pub struct CheckResult {
    pub name: &'static str,
    pub passed: bool,
    pub status: CheckStatus,
    pub detail: String,
}

pub enum CheckStatus {
    Ok,
    Warn,
    Fail,
}

impl std::fmt::Display for CheckStatus {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            CheckStatus::Ok => write!(f, "OK"),
            CheckStatus::Warn => write!(f, "WARN"),
            CheckStatus::Fail => write!(f, "FAIL"),
        }
    }
}

pub fn run_checks(server: &CortexMemServer) -> Vec<CheckResult> {
    let mut results = Vec::new();

    // 1. Binary version
    results.push(CheckResult {
        name: "Binary version",
        passed: true,
        status: CheckStatus::Ok,
        detail: format!("v{}", env!("CARGO_PKG_VERSION")),
    });

    // 2. Database accessible
    let stats = server.call_stats(None);
    match stats {
        Ok(s) => {
            results.push(CheckResult {
                name: "Database",
                passed: true,
                status: CheckStatus::Ok,
                detail: format!("{} observations", s.total),
            });
        }
        Err(e) => {
            results.push(CheckResult {
                name: "Database",
                passed: false,
                status: CheckStatus::Fail,
                detail: format!("Error: {e}"),
            });
        }
    }

    // 3. Schema version
    let mgr = server.memory_lock();
    let schema_version = mgr.db().get_meta("schema_version");
    match schema_version {
        Some(v) => {
            let ver: i64 = v.parse().unwrap_or(0);
            if ver >= CURRENT_SCHEMA_VERSION {
                results.push(CheckResult {
                    name: "Schema version",
                    passed: true,
                    status: CheckStatus::Ok,
                    detail: format!("v{ver}"),
                });
            } else {
                results.push(CheckResult {
                    name: "Schema version",
                    passed: false,
                    status: CheckStatus::Warn,
                    detail: format!("v{ver} (current: v{CURRENT_SCHEMA_VERSION})"),
                });
            }
        }
        None => {
            results.push(CheckResult {
                name: "Schema version",
                passed: false,
                status: CheckStatus::Fail,
                detail: "Not found".into(),
            });
        }
    }

    // 4. Embedding model
    match mgr.embed_mgr() {
        Some(e) => {
            let status = e.model_status();
            if status.downloaded {
                results.push(CheckResult {
                    name: "Embedding model",
                    passed: true,
                    status: CheckStatus::Ok,
                    detail: format!("{} ({}-dim)", status.model_name, status.dimensions),
                });
            } else {
                results.push(CheckResult {
                    name: "Embedding model",
                    passed: false,
                    status: CheckStatus::Warn,
                    detail: "Not downloaded. Run 'cortexmem model download'".into(),
                });
            }
        }
        None => {
            results.push(CheckResult {
                name: "Embedding model",
                passed: false,
                status: CheckStatus::Warn,
                detail: "Disabled (no cache directory)".into(),
            });
        }
    }

    // 5. FTS5 index count
    let fts_count = mgr.db().count_fts_entries();
    results.push(CheckResult {
        name: "FTS5 index",
        passed: fts_count > 0 || stats.as_ref().map(|s| s.total).unwrap_or(0) == 0,
        status: if fts_count > 0 || stats.as_ref().map(|s| s.total).unwrap_or(0) == 0 {
            CheckStatus::Ok
        } else {
            CheckStatus::Warn
        },
        detail: format!("{fts_count} entries indexed"),
    });

    // 6. Vector index count
    let vec_count = mgr.db().count_vector_entries();
    results.push(CheckResult {
        name: "Vector index",
        passed: true,
        status: CheckStatus::Ok,
        detail: format!("{vec_count} embeddings stored"),
    });

    drop(mgr);

    // 7-10: MCP config, cloud sync, git sync — checked via file existence
    results.extend(check_mcp_config());
    results.extend(check_sync_config());

    results
}

fn check_mcp_config() -> Vec<CheckResult> {
    let mut results = Vec::new();
    let home = match dirs::home_dir() {
        Some(h) => h,
        None => {
            results.push(CheckResult {
                name: "MCP config",
                passed: false,
                status: CheckStatus::Fail,
                detail: "Cannot determine home directory".into(),
            });
            return results;
        }
    };

    let agents = [
        ("Claude Code", home.join(".claude.json")),
        ("Cursor", home.join(".cursor").join("mcp.json")),
        ("Windsurf", home.join(".codeium").join("windsurf").join("mcp_config.json")),
        ("OpenCode", home.join(".config").join("opencode").join("config.json")),
        ("Gemini CLI", home.join(".gemini").join("settings.json")),
        ("Zed", home.join(".config").join("zed").join("settings.json")),
    ];

    let mut found = Vec::new();
    for (name, path) in &agents {
        if path.exists() {
            if let Ok(content) = std::fs::read_to_string(path) {
                if content.contains("cortexmem") {
                    found.push(*name);
                }
            }
        }
    }

    if found.is_empty() {
        results.push(CheckResult {
            name: "MCP config",
            passed: false,
            status: CheckStatus::Warn,
            detail: "No agent configured. Run 'cortexmem setup'".into(),
        });
    } else {
        results.push(CheckResult {
            name: "MCP config",
            passed: true,
            status: CheckStatus::Ok,
            detail: found.join(", "),
        });
    }

    results
}

fn check_sync_config() -> Vec<CheckResult> {
    let mut results = Vec::new();

    // Cloud sync
    let cloud_configured = std::env::var("CORTEXMEM_CLOUD_URL").is_ok();
    results.push(CheckResult {
        name: "Cloud sync",
        passed: cloud_configured,
        status: if cloud_configured { CheckStatus::Ok } else { CheckStatus::Warn },
        detail: if cloud_configured { "Configured".into() } else { "Not configured".into() },
    });

    // Git sync
    let home = dirs::home_dir().unwrap_or_default();
    let sync_dir = home.join(".cortexmem").join("sync");
    let git_configured = sync_dir.join(".git").exists();
    results.push(CheckResult {
        name: "Git sync",
        passed: git_configured,
        status: if git_configured { CheckStatus::Ok } else { CheckStatus::Warn },
        detail: if git_configured {
            format!("{}", sync_dir.display())
        } else {
            "Not configured".into()
        },
    });

    results
}

pub fn print_results(results: &[CheckResult]) {
    println!("cortexmem doctor v{}\n", env!("CARGO_PKG_VERSION"));

    let mut passed = 0;
    let mut failed = 0;
    let mut warned = 0;

    for r in results {
        let icon = match r.status {
            CheckStatus::Ok => "[OK]  ",
            CheckStatus::Warn => "[WARN]",
            CheckStatus::Fail => "[FAIL]",
        };
        println!("  {icon} {}: {}", r.name, r.detail);

        match r.status {
            CheckStatus::Ok => passed += 1,
            CheckStatus::Warn => warned += 1,
            CheckStatus::Fail => failed += 1,
        }
    }

    println!(
        "\n{passed} checks passed, {failed} failed, {warned} warnings"
    );
}
```

```rust
pub fn run_setup() -> Result<()> {
    println!("cortexmem setup\n");

    let detected = detect_installed_agents();
    let has_detected = detected.iter().any(|(_, d)| *d);

    if has_detected {
        println!("Detected agents:");
        for (i, (agent, det)) in detected.iter().enumerate() {
            if *det {
                println!("  [{}] {} (config found)", i + 1, agent);
            }
        }
        println!("\nNot detected:");
        for (agent, det) in &detected {
            if !*det {
                print!("  {agent}");
            }
        }
        println!("\n");
    }

    let items: Vec<String> = ALL_AGENTS.iter().map(|a| a.to_string()).collect();
    let selection = Select::new()
        .with_prompt("Which AI agent do you use?")
        .items(&items)
        .default(0)
        .interact()?;

    let agent = ALL_AGENTS[selection];
    println!("\nConfiguring cortexmem for {}...\n", agent);

    write_mcp_config(agent)?;

    if agent.supports_hooks() {
        println!("\nInstalling Claude Code plugin files...");
        install_claude_plugin()?;
    }

    // Post-setup verification
    println!("\nVerifying MCP server...");
    match verify_mcp_server() {
        Ok(tool_count) => println!("MCP server verified: {tool_count} tools available."),
        Err(e) => println!("Warning: Could not verify MCP server: {e}"),
    }

    println!("\nSetup complete! Restart {} to activate cortexmem.", agent);
    Ok(())
}
```
